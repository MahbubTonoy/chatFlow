CREATE TABLE msgs(
    ID int(11) UNIQUE AUTO_INCREMENT NOT NULL,
    sender_suname VARCHAR(100) NOT NULL,
    receiver_suname VARCHAR(100) NOT NULL,
    content MEDIUMTEXT NOT NULL,
    imgs VARCHAR(255) DEFAULT 'N/A' NOT NULL,
    stat TEXT DEFAULT 'N/A' NOT NULL,
    dateofmsg TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (ID)
    );
