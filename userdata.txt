CREATE TABLE userdata(
    ID int NOT NULL UNIQUE AUTO_INCREMENT,
    Username varchar(255) NOT NULL UNIQUE,
    SUname varchar(255) NOT NULL UNIQUE,
    XSUname varchar(255) NOT NULL UNIQUE,
    Fname varchar(255) NOT NULL,
    Lname varchar(255) NOT NULL,
    Fullname varchar(255) NOT NULL,
    Bio varchar(101) DEFAULT 'Hi. I am Using Chatflow!',
    Email varchar(255) NOT NULL UNIQUE,
    Pwd varchar(255) NOT NULL,
    Gender varchar(10) NOT NULL,
    DOB date NOT NULL,
    Signup_date timestamp NOT NULL,
    IP varchar(255) NOT NULL,
    DP varchar(255) DEFAULT 'N/A',
    Cover_Photo varchar(255) DEFAULT 'N/A',
    Stat varchar(255) DEFAULT 'N/A' NOT NULL,
    SecQ VARCHAR(30) NOT NULL DEFAULT 'N/A',
    SecA VARCHAR(30) NOT NULL DEFAULT 'N/A',
    RTN VARCHAR(10) NOT NULL DEFAULT 'true',
    NS VARCHAR(10) NOT NULL DEFAULT 'true',
    PRIMARY KEY (ID)
    );
